Getting Started with Die Rolling

Shi Fan Jin

2018-04-27

Die Rolling

The package ‘“dieroller”’ is a minimal implementation for simulating rolling a die a given number of times, and to visualize the relative frequencies in such a series of rolls.

Creating a ‘die’

The first step is to create a ‘“die”’ object with the function ‘die()’:

fair_die <- die()
fair_die
#> object "die"
#> 
#>   side      prob
#> 1    1 0.1666667
#> 2    2 0.1666667
#> 3    3 0.1666667
#> 4    4 0.1666667
#> 5    5 0.1666667
#> 6    6 0.1666667

By default, ‘die()’ creates a fair die with standard 6 sides and with probability values for each side of 1/6.

You can change the defaults with arguments sides and prob:

# create a loaded die
loaded_die <- die(prob = c(0.075, 0.1, 0.125, 0.15, 0.20, 0.35))
loaded_die
#> object "die"
#> 
#>   side  prob
#> 1    1 0.075
#> 2    2 0.100
#> 3    3 0.125
#> 4    4 0.150
#> 5    5 0.200
#> 6    6 0.350

#die with non-standard sides
weird_die <- die(sides = c('i', 'ii', 'iii', 'iv', 'v', 'vi'))
weird_die
#> object "die"
#> 
#>   side      prob
#> 1    i 0.1666667
#> 2   ii 0.1666667
#> 3  iii 0.1666667
#> 4   iv 0.1666667
#> 5    v 0.1666667
#> 6   vi 0.1666667

Rolling a Die

Once you have defined a ‘“die”’ object, you can use the function ‘roll()’ to roll a die a given number of ‘times’. The output will be an object of class ‘“roll”’, which will contain the vector of ‘rolls’:

# roll fair die 50 times
fair50 <- roll(fair_die, times = 50)
fair50
#> object "roll"
#> 
#> $rolls
#>  [1] 2 6 3 4 5 2 2 1 5 6 1 6 1 2 1 1 6 4 2 4 3 1 4 3 6 1 4 6 2 3 1 6 6 4 5
#> [36] 6 1 4 2 5 6 2 3 5 6 1 4 5 6 1

# roll 50 times
set.seed(123)
fair_50rolls <- roll(fair_die, times = 50)
# print
fair_50rolls
#> object "roll"
#> 
#> $rolls
#>  [1] 3 6 4 1 1 2 5 1 5 4 1 4 6 5 2 1 3 2 3 1 1 6 5 1 5 6 5 5 3 2 1 1 6 6 2
#> [36] 4 6 3 3 3 2 4 4 4 2 2 3 4 3 1

# what's in fair50?
names(fair50)
#> [1] "rolls" "sides" "prob"  "total"
fair50$rolls
#>  [1] 2 6 3 4 5 2 2 1 5 6 1 6 1 2 1 1 6 4 2 4 3 1 4 3 6 1 4 6 2 3 1 6 6 4 5
#> [36] 6 1 4 2 5 6 2 3 5 6 1 4 5 6 1
fair50$sides
#> [1] 1 2 3 4 5 6
fair50$prob
#> [1] 0.1666667 0.1666667 0.1666667 0.1666667 0.1666667 0.1666667
fair50$total
#> [1] 50

# string die
str_die <- die(
  sides = c('a', 'b', 'c', 'd', 'e', 'f'),
  prob = c(0.075, 0.1, 0.125, 0.15, 0.20, 0.35))
# roll 20 times
set.seed(123)
str_rolls <- roll(str_die, times = 20)
names(str_rolls)
#> [1] "rolls" "sides" "prob"  "total"
str_rolls
#> object "roll"
#> 
#> $rolls
#>  [1] "f" "c" "e" "b" "a" "f" "e" "b" "d" "e" "a" "e" "d" "d" "f" "b" "f"
#> [18] "f" "f" "a"

The ‘summary()’ function gives you a nice table of frequencies:

set.seed(123)
fair50_sum <- summary(fair_50rolls)
fair50_sum
#> summary "roll"
#> 
#>   side count prob
#> 1    1    11 0.22
#> 2    2     8 0.16
#> 3    3     9 0.18
#> 4    4     8 0.16
#> 5    5     7 0.14
#> 6    6     7 0.14

# what's in the summary
class(fair50_sum)
#> [1] "summary.roll"
names(fair50_sum)
#> [1] "freqs"
fair50_sum$freqs
#>   side count prob
#> 1    1    11 0.22
#> 2    2     8 0.16
#> 3    3     9 0.18
#> 4    4     8 0.16
#> 5    5     7 0.14
#> 6    6     7 0.14

Plot of relative frequencies

You can use the function ‘plot()’ to create a graohic with the relative frequencies of a series of die rolls:

plot(fair_50rolls)

Additional Method

There are also some additional method of using the object ‘die’ and the function ‘roll()’, such as an extraction value for the given roll, a replacement method to replace a given value of a given roll, and addition method to add more rolls.

# roll fair die
set.seed(123)
fair_die <- die()
fair500 <- roll(fair_die, times = 500)
# summary method
summary(fair500)
#> summary "roll"
#> 
#>   side count  prob
#> 1    1    80 0.160
#> 2    2    81 0.162
#> 3    3    92 0.184
#> 4    4    92 0.184
#> 5    5    72 0.144
#> 6    6    83 0.166

# extracting roll in position 500
fair500[500]
#> [1] 6

# replacing last roll
fair500[500] <- 1
fair500[500]
#> [1] 1
#checking
summary(fair500)
#> summary "roll"
#> 
#>   side count  prob
#> 1    1    81 0.162
#> 2    2    81 0.162
#> 3    3    92 0.184
#> 4    4    92 0.184
#> 5    5    72 0.144
#> 6    6    82 0.164

# adding 100 rolls
fair600 <- fair500 + 100
#checking
summary(fair600)
#> summary "roll"
#> 
#>   side count      prob
#> 1    1   100 0.1666667
#> 2    2    97 0.1616667
#> 3    3   104 0.1733333
#> 4    4   109 0.1816667
#> 5    5    91 0.1516667
#> 6    6    99 0.1650000

# plot method
plot(fair500, 500)
